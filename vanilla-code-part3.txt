    <script>
        const order = JSON.parse(localStorage.getItem('rikyPharmaOrder'));
        let selectedMethod = 'upi';

        if (!order) {
            window.location.href = 'cart.html';
        }

        document.getElementById('billNumber').textContent = `Bill Number: ${order.billNumber}`;

        // Render Bill Details
        document.getElementById('billDetails').innerHTML = `
            <div class="detail-row">
                <span>Bill Number:</span>
                <strong>${order.billNumber}</strong>
            </div>
            <div class="detail-row">
                <span>Date & Time:</span>
                <strong>${new Date(order.date).toLocaleString('en-IN')}</strong>
            </div>
            <div class="detail-row">
                <span>Customer Name:</span>
                <strong>${order.user.name}</strong>
            </div>
            <div class="detail-row">
                <span>Mobile Number:</span>
                <strong>${order.user.mobile}</strong>
            </div>
        `;

        // Render Bill Items
        document.getElementById('billItems').innerHTML = `
            <table>
                <thead>
                    <tr>
                        <th>#</th>
                        <th>Product</th>
                        <th>Qty</th>
                        <th>Price</th>
                        <th>Total</th>
                    </tr>
                </thead>
                <tbody>
                    ${order.items.map((item, index) => `
                        <tr>
                            <td>${index + 1}</td>
                            <td>
                                <strong>${item.name}</strong><br>
                                <small>${item.composition}</small>
                            </td>
                            <td>${item.quantity}</td>
                            <td>₹${item.mrp.toFixed(2)}</td>
                            <td>₹${(item.mrp * item.quantity).toFixed(2)}</td>
                        </tr>
                    `).join('')}
                </tbody>
            </table>
        `;

        // Render Bill Total
        const totalItems = order.items.reduce((sum, item) => sum + item.quantity, 0);
        document.getElementById('billTotal').innerHTML = `
            <div class="total-row">
                <span>Total Items:</span>
                <span>${totalItems}</span>
            </div>
            <div class="total-row">
                <span>Subtotal:</span>
                <span>₹${order.total.toFixed(2)}</span>
            </div>
            <div class="total-row">
                <span>Tax (Included):</span>
                <span>₹0.00</span>
            </div>
            <div class="total-row grand-total">
                <span>Grand Total:</span>
                <span>₹${order.total.toFixed(2)}</span>
            </div>
        `;

        function selectPayment(method) {
            selectedMethod = method;
            document.querySelectorAll('.payment-method').forEach(btn => {
                btn.classList.remove('active');
                if (btn.dataset.method === method) {
                    btn.classList.add('active');
                }
            });
            renderPaymentDetails();
        }

        function renderPaymentDetails() {
            const details = document.getElementById('paymentDetails');

            if (selectedMethod === 'upi') {
                const upiId = 'crramkumar1976-2@okicici';
                const upiString = `upi://pay?pa=${upiId}&pn=Riky Pharma&am=${order.total.toFixed(2)}&cu=INR`;

                details.innerHTML = `
                    <div class="payment-info">
                        <h3>Scan QR Code to Pay</h3>
                        <div id="qrcode" class="qr-code"></div>
                        <p>UPI ID: ${upiId}</p>
                        <p class="amount-large">₹${order.total.toFixed(2)}</p>
                    </div>
                `;

                // Generate QR Code
                new QRCode(document.getElementById("qrcode"), {
                    text: upiString,
                    width: 200,
                    height: 200
                });
            } else if (selectedMethod === 'cash') {
                details.innerHTML = `
                    <div class="payment-info">
                        <div class="payment-icon-large">💵</div>
                        <h3>Cash Payment</h3>
                        <p>Amount to be paid in cash</p>
                        <p class="amount-large">₹${order.total.toFixed(2)}</p>
                    </div>
                `;
            } else if (selectedMethod === 'card') {
                details.innerHTML = `
                    <div class="payment-info">
                        <div class="payment-icon-large">💳</div>
                        <h3>Card Payment</h3>
                        <p>Swipe or insert your card</p>
                        <p class="amount-large">₹${order.total.toFixed(2)}</p>
                    </div>
                `;
            }
        }

        function confirmPayment() {
            localStorage.removeItem('rikyPharmaCart');
            window.location.href = 'success.html';
        }

        function downloadPDF() {
            const { jsPDF } = window.jspdf;
            html2canvas(document.getElementById('billContent')).then(canvas => {
                const imgData = canvas.toDataURL('image/png');
                const pdf = new jsPDF('p', 'mm', 'a4');
                const pdfWidth = pdf.internal.pageSize.getWidth();
                const pdfHeight = (canvas.height * pdfWidth) / canvas.width;
                pdf.addImage(imgData, 'PNG', 0, 0, pdfWidth, pdfHeight);
                pdf.save(`Riky_Pharma_Bill_${order.billNumber}.pdf`);
            });
        }

        renderPaymentDetails();
    </script>
</body>
</html>
